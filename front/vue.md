# install nodejs
	1. get nodejs 8 to apt
		curl -sL https://deb.nodesource.com/setup_8.x | sudo -E bash -
	2. install nodejs 8
		apt install nodejs
	3. node -v
		v8.5.0

# install taobao mirror
	1. npm install -g cnpm --registry=https://registry.npm.taobao.org
	2. npm i -g npm
